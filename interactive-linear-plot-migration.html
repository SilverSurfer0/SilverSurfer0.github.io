df = pd.read_csv('Uusimaa_1990to2016_migration.csv', header=1, encoding='cp1252')

#Adding columns with total(sum) values across ages groups
#Getting list of colnames relative to certain groups
colnames = df.columns.values
cols_immigration = []
cols_emigration = []
cols_netmigration = []
for name in colnames:
    if name.startswith('Immigration '): cols_immigration.append(name)
    elif name.startswith('Emigration '): cols_emigration.append(name)
    elif name.startswith('Net migration '): cols_netmigration.append(name)

#Adding Immigration column with total values across ages groups
immigration_total = np.sum(df[cols_immigration],axis=1)
df['Immigration'] = immigration_total

#Adding Emigration column with total values across ages groups
emigration_total = np.sum(df[cols_emigration],axis=1)
df['Emigration'] = emigration_total

#Adding Net Migration column with total values across ages groups
net_migration_total = np.sum(df[cols_netmigration],axis=1)
df['Net Migration'] = net_migration_total

#Separate dataframes into dataframe with 1) Total values for all counriesonly and 2)dataframe with values for each country
df_countries_total = df[df['Country of exit and entry']=='COUNTRIES, TOTAL']
df_countries = df[df['Country of exit and entry']!='COUNTRIES, TOTAL']

### Event on moving the cursor    
def on_move(event):
    
    visibility_changed = False
    linewidth, color, alpha = 1,'white',0
    pe_linewidth, pe_color, pe_alpha  = 0,'white',0
    
    for idx, (item, ann_text,annotation) in enumerate(lines_with_annotation_text):
                
        should_be_visible = (item.contains(event)[0] == True)
        
        if should_be_visible:
            if isinstance(item,linetype):
                linewidth, color, alpha = line_params[idx]['linewidth']*1.1,line_params[idx]['color'],1
                pe_linewidth, pe_color, pe_alpha = line_params[idx]['linewidth']*1.2,'black',0.8
            elif isinstance(item,collectiontype):
                plt.setp(item, alpha=1, edgecolor='#A74100', linewidth=0.4)
                idx=1
                linewidth, color, alpha = line_params[idx]['linewidth']*1.1,line_params[idx]['color'],1
                pe_linewidth, pe_color, pe_alpha = line_params[idx]['linewidth']*1.2,'black',0.8
                
        else:
            if isinstance(item,linetype):
                linewidth, color, alpha = line_params[idx]['linewidth'],line_params[idx]['color'],line_params[idx]['alpha']
                pe_linewidth, pe_color, pe_alpha = 0,'white',0
            elif isinstance(item,collectiontype):
                plt.setp(item, alpha=0.6, edgecolor='#A74100', linewidth=0.3 )
                idx=1
                linewidth, color, alpha = line_params[idx]['linewidth'],line_params[idx]['color'],line_params[idx]['alpha']
                pe_linewidth, pe_color, pe_alpha = 0,'white',0

        if should_be_visible != ann_text.get_visible():
            visibility_changed = True
            
            xpos, ypos = event.xdata, event.ydata
            xdata = plt.getp(item,'data')[0]
            ydata = plt.getp(item,'data')[1]
            indx = np.searchsorted(xdata, [xpos])[0]
            x = xdata[indx]
            y = ydata[indx]
            item_label = plt.getp(item,'label')
                             
            ann_text.set_text(item_label)
            ann_text.set_visible(should_be_visible)
                             
            annotation.set_text(str(int(y))+' people in '+str(int(x)))
            annotation.set_position((x+0.2,y))
            annotation.set_visible(should_be_visible)
            annotation.set_verticalalignment('top')
            
            if isinstance(item,linetype):
                plt.setp(item, linewidth=linewidth, color=color, alpha=alpha)
                plt.setp(item, path_effects=[path_effects.Stroke(linewidth=pe_linewidth, foreground=pe_color, alpha=pe_alpha),path_effects.Normal()])
            elif isinstance(item,collectiontype):
                plt.setp(item, alpha=1, edgecolor='#A74100', linewidth=0.4)
                plt.setp(lines_with_annotation_text[1], linewidth=linewidth, color=color, alpha=alpha)
                
                
    if visibility_changed:        
        plt.draw()

on_move_id = fig.canvas.mpl_connect('motion_notify_event', on_move)

### Graph Tuning and Configuration

def tune_graph():
    # Getting min and max for x and y axes
    xmin, xmax = ax.get_xlim()
    ax.set_xlim(xmin, xmax) #Setting xmin and xmax limits in stone
    ymin,ymax = ax.get_ylim()

    #Get rid of unnecessary spines
    sns.despine(left=True,bottom=True)
    for spine in ax.spines.values():
        spine.set_color('grey')

    #Set tick configuration
    plt.tick_params(axis='both', top='off', bottom='off', left='on', right='off', labelleft='on', labelbottom='on', color='#404040',labelcolor='#404040')


    #plot horizontal lines
    ax.hlines(0, xmin-0.5, xmax-0.7, linestyle='--', linewidth=0.8, alpha=0.8)
    ax.hlines(-5000, xmin-0.5, xmax-0.6, linestyle='--', linewidth=0.6, alpha=0.3)
    ax.hlines(5000, xmin-0.5, xmax-0.6, linestyle='--', linewidth=0.6, alpha=0.3)
    ax.hlines(10000, xmin-0.5, xmax-0.6, linestyle='--', linewidth=0.6, alpha=0.3)

    #Titles, annotations
    ax.set_title('Migration of 15-44 y/olds in Uusimaa, Finland Region', alpha=0.8, fontsize = 12)

    #Add annotations separately for each line
    ann_text = ax.text(xmax*0.9998,
             list(df_countries_total['Immigration'])[-1],
             'Immigration',
            fontsize = 10, color='#E00003',alpha=0.9)
    ann_text = ax.text(xmax*0.9998,
             list(df_countries_total['Net Migration'])[-1],
             'Net\nMigration',
            fontsize = 10, color='#FF6300')
    ann_text = ax.text(xmax*0.9998,
             list(df_countries_total['Emigration'])[-1],
             'Emigration',
            fontsize = 10, color='#0070D6',alpha=0.9)


    ax.set_ylabel('People')
    ax.set_xlabel('Year')
